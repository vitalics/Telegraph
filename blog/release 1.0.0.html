Installation​

Button​

pnpm:

pnpm add @tlgr/button@1.0.0

yarn:

yarn add @tlgr/button@1.0.0

npm:

npm i @tlgr/button@1.0.0

Date Picker​

pnpm:

pnpm add @tlgr/date-picker@1.0.0

yarn:

yarn add @tlgr/date-picker@1.0.0

npm:

npm i @tlgr/date-picker@1.0.0
